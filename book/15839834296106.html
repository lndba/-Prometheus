数据拉取配置

  scrape_configs主要用于配置拉取数据节点，每一个拉取配置模板如下：

job_name: <job_name> #任务名称，必须唯一
[ scrape_interval: <duration> | default = <global_config.scrape_interval> ] #拉取时间间隔，默认使用全局配置
[ scrape_timeout: <duration> | default = <global_config.scrape_timeout> ] #拉取超时时间，默认使用全局配置
[ metrics_path: <path> | default = /metrics ] #拉取节点的metric路径，默认是/metrics
[ honor_labels: <boolean> | default = false ] #用于解决拉取数据标签有冲突，当设置为 true, 以拉取数据为准，否则以服务配置为准，默认是false
[ scheme: <scheme> | default = http ] #拉取数据访问协议
params: #数据拉取访问时带的http请求参数
  [ <string>: [<string>, ...] ]
basic_auth: #指定认证的信息
  [ username: <string> ]
  [ password: <string> ]
[ bearer_token: <string> ]
[ bearer_token_file: /path/to/bearer/token/file ]
tls_config: #配置CA证书
  [ <tls_config> ]
[ proxy_url: <string> ] #配置以代理的形式访问数据拉取目标
openstack_sd_configs: #OpenStack服务发现
  [ - <openstack_sd_config> ... ]
file_sd_configs: #文件服务发现
  [ - <file_sd_config> ... ]
kubernetes_sd_configs: #Kubernetes 服务发现
  [ - <kubernetes_sd_config> ... ]
static_configs: #静态配置
  [ - <static_config> ... ]
relabel_configs: #拉取数据重置标签配置，可以在抓取目标之前动态重写目标的标签，可以配置多个，按照先后顺序应用
  [ - <relabel_config> ... ]
metric_relabel_configs: #metric重置标签配置
  [ - <relabel_config> ... ]
[ sample_limit: <int> | default = 0 ] #存储的数据标签个数限制，如果超过限制，该数据将被忽略；默认值为0，表示没有限制
