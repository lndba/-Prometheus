通过node_exporter监控Linux主机

  Prometheus Node exporter，用于由* NIX内核公​​开的硬件和操作系统metrics，用Go编写，带有可插入的metrics收集器。
官方参考：
https://prometheus.io/docs/instrumenting/exporters/
https://github.com/prometheus/node_exporter

安装node_exporter

  为监控服务器CPU、内存、磁盘、I/O等信息，首先需要安装node_exporter。node_exporter的作用是用于机器系统数据收集。

#下载node_exporter
[root@prometheus-server sd_configs]# cd /usr/local/src/
[root@prometheus-server src]# wget https://github.com/prometheus/node_exporter/releases/download/v0.18.1/node_exporter-0.18.1.linux-amd64.tar.gz

#解压node_exporter
[root@prometheus-server src]# tar zxvf node_exporter-0.18.1.linux-amd64.tar.gz 

#将解压后的目录移动到/usr/local/目录下
[root@prometheus-server src]# mv node_exporter-0.18.1.linux-amd64 /usr/local/node_exporter

#查看node_exporter目录下的文件
[root@prometheus-server src]# cd /usr/local/node_exporter
[root@prometheus-server node_exporter]# ls

#查看node_exporter版本
[root@prometheus-server node_exporter]# ./node_exporter --version
node_exporter, version 0.18.1 (branch: HEAD, revision: 3db77732e925c08f675d7404a8c46466b2ece83e)
  build user:       root@b50852a1acba
  build date:       20190604-16:41:18
  go version:       go1.12.5

#创建systemd服务单元文件管理node_exporter
[root@prometheus-server node_exporter]# cat > /usr/lib/systemd/system/node_exporter.service << 'EOF'
[Unit]
Description=Node Exporter
Wants=network-online.target
After=network-online.target

[Service]
ExecStart=/usr/local/node_exporter/node_exporter

[Install]
WantedBy=default.target
EOF

#重新加载systemd服务
[root@prometheus-server node_exporter]# systemctl daemon-reload

#启动node_exporter
[root@prometheus-server node_exporter]# systemctl start node_exporter.service

#查看node_exporter状态
[root@prometheus-server node_exporter]# systemctl status node_exporter.service

#将node_exporter加入开机启动
[root@prometheus-server node_exporter]# systemctl enable node_exporter.service

#浏览器访问：http://192.168.3.11:9100/metrics

修改prometheus.yml文件

  - job_name: 'node_exporter'
    file_sd_configs:
      - files:
        - sd_configs/nodes.yml
        refresh_interval: 5s

创建sd_configs/nodes.yml文件

[root@prometheus-server prometheus]# mkdir /usr/local/prometheus/sd_configs
[root@prometheus-server prometheus]# cd sd_configs/
[root@prometheus-server sd_configs]# cat nodes.yml 
- targets:
  - 192.168.3.11:9100
  labels:
    region: cn-beijing
    service: nginx

- targets:
  - 192.168.3.12:9100
  labels:
    region: cn-beijing
    service: mysql
