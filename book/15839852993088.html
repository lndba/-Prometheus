部署Alertmanager

#下载alertmanager
[root@prometheus-server ~]# cd /usr/local/src/
[root@prometheus-server src]# wget https://github.com/prometheus/alertmanager/releases/download/v0.20.0/alertmanager-0.20.0.linux-amd64.tar.gz

#解压alertmanager
[root@prometheus-server src]# tar zxvf alertmanager-0.20.0.linux-amd64.tar.gz

#将解压后的目录移动到/usr/local/目录下
[root@prometheus-server src]# mv alertmanager-0.20.0.linux-amd64 /usr/local/alertmanager
[root@prometheus-server src]# cd /usr/local/alertmanager/

#查看安装版本
[root@prometheus-server alertmanager]# ./alertmanager --version
alertmanager, version 0.20.0 (branch: HEAD, revision: f74be0400a6243d10bb53812d6fa408ad71ff32d)
  build user:       root@00c3106655f8
  build date:       20191211-14:13:14
  go version:       go1.13.5

Alertmanager配置文件详解

[root@prometheus-server alertmanager]# cat alertmanager.yml 
global: #全局配置
  resolve_timeout: 5m #解决告警时间间隔，判断告警是否解决，然后发送通知

route: #定义告警分发的规则
  group_by: ['alertname'] #告警分组名称
  group_wait: 10s #等待多长时间发送一组告警通知
  group_interval: 10s #在发送新警报前的等待时间
  repeat_interval: 1h #发送重复告警的时间
  receiver: 'web.hook' #接受者名称，与receivers的name一致

receivers: #定义告警接受者信息
- name: 'web.hook'
  webhook_configs:
  - url: 'http://127.0.0.1:5001/'

inhibit_rules: #定义抑制的规则
  - source_match:
      severity: 'critical'
    target_match:
      severity: 'warning'
    equal: ['alertname', 'dev', 'instance']

resolve_timeout：简单说就是在报警恢复的时候不是立马发送的，在接下来的这个时间内，如果没有此报警信息触发，才发送报警恢复消息。 默认值为5m。

配置alertmanager：

global:
  resolve_timeout: 5m
  smtp_from: 'wang_alert@163.com'
  smtp_smarthost: 'smtp.163.com:25'
  smtp_auth_username: 'wang_alert@163.com'
  smtp_auth_password: 'alertmanager123'
  smtp_require_tls: false

route:
  group_by: ['alertname']
  group_wait: 10s
  group_interval: 10s
  repeat_interval: 1m
  receiver: 'email'

receivers:
- name: 'email'
  email_configs:
  - to: 'kissinglm@163.com'
    send_resolved: true

检查配置文件是否正确：

[root@prometheus-server alertmanager]# ./amtool check-config ./alertmanager.yml

alertmanager使用systemd的方式启动

cat > /usr/lib/systemd/system/alertmanager.service << 'EOF'
[Unit]
Description=Alertmanager
Wants=network-online.target
After=network-online.target

[Service]
Type=simple
ExecStart=/usr/local/alertmanager/alertmanager \
    --config.file=/usr/local/alertmanager/alertmanager.yml

[Install]
WantedBy=multi-user.target
EOF

启动node_exporter服务

 #重新加载systemd服务
[root@prometheus-server alertmanager]# systemctl daemon-reload

#启动alertmanager
[root@prometheus-server alertmanager]# systemctl start alertmanager.service

#查看alertmanager状态
[root@prometheus-server alertmanager]# systemctl status alertmanager.service

#将alertmanager加入开机启动
[root@prometheus-server alertmanager]# systemctl enable alertmanager.service
