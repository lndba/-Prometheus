Prometheus二进制方式安装

环境准备

主机名 IP地址 角色 系统版本
prometheus-server 192.168.3.11 server CentOS7.7
linux-node1 192.168.3.12 node CentOS7.7 
#将下载的包放到这个目录下
[root@prometheus-server ~]# cd /usr/local/src/

#下载prometheus
[root@prometheus-server src]# wget https://github.com/prometheus/prometheus/releases/download/v2.15.2/prometheus-2.15.2.linux-amd64.tar.gz

#解压prometheus二进制包
[root@prometheus-server src]# tar zxvf prometheus-2.15.2.linux-amd64.tar.gz
[root@prometheus-server src]# ls
prometheus-2.15.2.linux-amd64  prometheus-2.15.2.linux-amd64.tar.gz

#将解压后的目录移动到/usr/local/目录下
[root@prometheus-server src]# mv prometheus-2.15.2.linux-amd64 /usr/local/prometheus

#进入prometheus目录查看目录下有哪些文件
[root@prometheus-server src]# cd /usr/local/prometheus
[root@prometheus-server prometheus]# ls

#将prometheus的二进制文件软连接到/usr/local/bin下
[root@prometheus-server prometheus]# ln -s /usr/local/prometheus/prometheus /usr/local/bin/prometheus
[root@prometheus-server prometheus]# ln -s /usr/local/prometheus/promtool /usr/local/bin/promtool

#创建prometheus时序数据库存储目录
[root@prometheus-server prometheus]# mkdir -p /data/prometheus/tsdb

#查看prometheus版本
[root@prometheus-server prometheus]# prometheus --version
prometheus, version 2.15.2 (branch: HEAD, revision: d9613e5c466c6e9de548c4dae1b9aabf9aaf7c57)
  build user:       root@688433cf4ff7
  build date:       20200106-14:50:51
  go version:       go1.13.5

#prometheus启动参数介绍
[root@prometheus-server prometheus]# prometheus --help
--config.file="prometheus.yml" #指定配置文件
--web.listen-address="0.0.0.0:9090" #监听地址
--storage.tsdb.path="data/" #tsdb默认存储的目录
--storage.tsdb.retention=STORAGE.TSDB.RETENTION #这个参数已经被弃用了
--storage.tsdb.retention.time=STORAGE.TSDB.RETENTION.TIME #何时删除旧数据，默认为15d
--web.enable-lifecycle #启用远程热加载配置文件调用指令是curl -X POST http://localhost:9090/-/reload

#prometheus配置文件默认监控9090端口
[root@prometheus-server prometheus]# less prometheus.yml

#使用promtool检测配置文件语法
[root@prometheus-server prometheus]# promtool check config /usr/local/prometheus/prometheus.yml     

#手动启动
[root@prometheus-server prometheus]# prometheus --config.file=/usr/local/prometheus/prometheus.yml --storage.tsdb.path=/data/prometheus/tsdb
